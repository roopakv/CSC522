				\textbf{Ans:} \\
				2nd column alone : x[,2] $\implies$ [1] 4 5 6\\
				3rd column alone : x[,3] $\implies$ [1] 7 8 9\\

				$\implies$ \\
				
				$\begin{array}{ccc} 
					~		&	[,1]	&	[,2]	\\ \relax

matrixsum <- function (mat) {
  i<-1
  sum<-0
  while(i<=dim(mat)[1]*dim(mat)[2])
  {
    sum<-sum+mat[i]
    i<-i+1
  }
  print (paste(sum, "is the sum of elements"))
}
